    "## 2. Memory-Efficient Data Loading (Chunked Sampling)\n",
    "\n",
    "The full dataset is ~3.5 GB (~100M rows). We load it in chunks of 5M rows,\n",
    "sample from each chunk, and discard the rest. This keeps RAM under ~2 GB."
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "# Column names for the dataset\n",
    "COL_NAMES = ['user_id', 'item_id', 'category_id', 'behavior_type', 'timestamp']\n",
    "COL_DTYPES = {\n",
    "    'user_id': 'int32',\n",
    "    'item_id': 'int32',\n",
    "    'category_id': 'int32',\n",
    "    'behavior_type': 'category',\n",
    "    'timestamp': 'int64'\n",
    "}\n",
    "\n",
    "CHUNK_SIZE = 5_000_000  # 5M rows per chunk\n",
    "SAMPLE_TARGET = 1_000_000  # Total target sample size\n",
    "\n",
    "print('Loading data in chunks (this takes 3-5 minutes)...')\n",
    "print('=' * 60)\n",
    "\n",
    "sampled_chunks = []\n",
    "total_rows = 0\n",
    "behavior_counts_full = {'pv': 0, 'buy': 0, 'cart': 0, 'fav': 0}\n",
    "\n",
    "start_time = time.time()\n",
    "\n",
    "for i, chunk in enumerate(pd.read_csv(\n",
    "    DATA_PATH,\n",
    "    header=None,\n",
    "    names=COL_NAMES,\n",
    "    dtype=COL_DTYPES,\n",
    "    chunksize=CHUNK_SIZE\n",
    ")):\n",
    "    total_rows += len(chunk)\n",
    "\n",
    "    # Count behaviors in full dataset\n",
    "    counts = chunk['behavior_type'].value_counts()\n",
    "    for btype in behavior_counts_full:\n",
    "        if btype in counts.index:\n",
    "            behavior_counts_full[btype] += counts[btype]\n",
    "\n",
    "    # Stratified sampling: oversample 'buy' events\n",
    "    buy_mask = chunk['behavior_type'] == 'buy'\n",
    "    buy_rows = chunk[buy_mask]\n",
    "    non_buy_rows = chunk[~buy_mask]\n",
    "\n",
    "    # Sample proportionally, but keep ALL buy events\n",
    "    sample_size = min(SAMPLE_TARGET // 20, len(non_buy_rows))  # ~50K non-buy per chunk\n",
    "    sampled = pd.concat([\n",
    "        buy_rows,\n",
    "        non_buy_rows.sample(n=sample_size, random_state=42)\n",
    "    ])\n",
    "    sampled_chunks.append(sampled)\n",
    "\n",
    "    elapsed = time.time() - start_time\n",
    "    print(f'  Chunk {i+1}: {total_rows:>12,} rows processed | '\n",
    "          f'{len(sampled):>7,} sampled | {elapsed:.0f}s elapsed')\n",
    "\n",
    "    # Free memory\n",
    "    del chunk, buy_rows, non_buy_rows, sampled\n",
    "    gc.collect()\n",
    "\n",
    "# Combine all sampled chunks\n",
    "df = pd.concat(sampled_chunks, ignore_index=True)\n",
    "del sampled_chunks\n",
    "gc.collect()\n",
    "\n",
    "# If we have more than target, downsample non-buy\n",
    "if len(df) > SAMPLE_TARGET:\n",
    "    buy_df = df[df['behavior_type'] == 'buy']\n",
    "    non_buy_df = df[df['behavior_type'] != 'buy'].sample(\n",
    "        n=SAMPLE_TARGET - len(buy_df), random_state=42\n",
    "    )\n",
    "    df = pd.concat([buy_df, non_buy_df], ignore_index=True)\n",
    "    del buy_df, non_buy_df\n",
    "    gc.collect()\n",
    "\n",
    "elapsed_total = time.time() - start_time\n",
    "print('\\n' + '=' * 60)\n",
    "print(f'✅ Data loading complete in {elapsed_total:.0f} seconds')\n",
    "print(f'   Full dataset: {total_rows:,} rows')\n",
    "print(f'   Sampled dataset: {len(df):,} rows')\n",
    "print(f'   Memory usage: {df.memory_usage(deep=True).sum() / 1024**2:.1f} MB')\n",
    "print(f'\\n📊 Full dataset behavior counts:')\n",
    "for btype, count in sorted(behavior_counts_full.items(), key=lambda x: -x[1]):\n",
    "    pct = count / total_rows * 100\n",
    "    print(f'   {btype:>5}: {count:>12,} ({pct:.2f}%)')"

    "## 5. Feature Engineering"

   "cell_type": "code",

   "source": [
    "print('Engineering features per user...')\n",
    "start = time.time()\n",
    "\n",
    "# Label: Did the user make ANY purchase?\n",
    "user_purchased = df[df['behavior_type'] == 'buy']['user_id'].unique()\n",
    "user_label = pd.DataFrame({'user_id': df['user_id'].unique()})\n",
    "user_label['purchased'] = user_label['user_id'].isin(user_purchased).astype(int)\n",
    "\n",
    "print(f'  Users who purchased: {len(user_purchased):,} ({len(user_purchased)/len(user_label)*100:.2f}%)')\n",
    "print(f'  Users who did NOT purchase: {len(user_label) - len(user_purchased):,}')\n",
    "\n",
    "# Group by user\n",
    "user_groups = df.groupby('user_id')\n",
    "\n",
    "# Feature 1: Total interaction count\n",
    "feat_total = user_groups.size().rename('total_interactions')\n",
    "\n",
    "# Feature 2: Behavior counts\n",
    "feat_behavior = df.groupby(['user_id', 'behavior_type']).size().unstack(fill_value=0)\n",
    "feat_behavior.columns = [f'count_{c}' for c in feat_behavior.columns]\n",
    "\n",
    "# Feature 3: Cart addition frequency\n",
    "feat_cart_freq = feat_behavior.get('count_cart', pd.Series(0, index=feat_behavior.index)).rename('cart_freq')\n",
    "\n",
    "# Feature 4: Favorite-to-cart ratio\n",
    "fav_col = feat_behavior.get('count_fav', pd.Series(0, index=feat_behavior.index))\n",
    "cart_col = feat_behavior.get('count_cart', pd.Series(0, index=feat_behavior.index))\n",

    "# ── LSTM Model Definition ──\n",
    "class PurchaseLSTM(nn.Module):\n",
    "    def __init__(self, vocab_size=5, embed_dim=32, hidden_dim=64,\n",
    "                 num_layers=2, dropout=0.2):\n",
    "        super().__init__()\n",
    "        self.embedding = nn.Embedding(vocab_size, embed_dim, padding_idx=0)\n",
    "        self.lstm = nn.LSTM(embed_dim, hidden_dim, num_layers=num_layers,\n",
    "                           batch_first=True, dropout=dropout)\n",
    "        self.dropout = nn.Dropout(dropout)\n",
    "        self.fc = nn.Linear(hidden_dim, 1)\n",
    "\n",
    "    def forward(self, x):\n",
    "        embedded = self.embedding(x)\n",
    "        lstm_out, (hidden, _) = self.lstm(embedded)\n",
    "        out = self.dropout(hidden[-1])  # Use last hidden state\n",
    "        return self.fc(out).squeeze(1)\n",
    "\n",
